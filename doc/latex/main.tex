\section{Aufgabenstellung}

\begin{frame}{Aufgabenstellung}
\begin{itemize}
    \item Design einer \textbf{Producer-Consumer-Applikation} für Eventaustausch
    \item Producer und Consumer laufen in \textbf{getrennten Prozessen}
    \item \textbf{Kommunikationsschicht muss austauschbar sein} (modulares Design)
    \item \textbf{IPC mit Systemfunktionen}: Windows oder Linux
    \item Implementierung in \textbf{C++17}, kompilierbarer Source Code
    \vspace{0.5em}
    \item \textbf{Producer} generiert Events: Zeitstempel, Komponente, Severity, Nachricht
    \item \textbf{Consumer} filtert ab \texttt{WARNING}, aggregiert und gibt Statistiken aus
\end{itemize}
\end{frame}

\section{Designentscheidungen}

\begin{frame}{Designentscheidungen -- Interfaces \& Patterns}
\begin{itemize}
    \item \textbf{Abstract Factory Pattern}\\
          \texttt{IIPCFactory} $\rightarrow$ \texttt{PipeIPCFactory}\\
          Transport per \texttt{--transport pipe} wählbar -- neue Impl.\ ohne Änderung an Business Logic

    \vspace{0.4em}
    \item \textbf{Interface Segregation}\\
          \texttt{IIPCWriter}, \texttt{IIPCReader}, \texttt{IIPCFactory}, \texttt{IEventSerializer}, \texttt{IStopper}\\
          Klassen kennen nur die Schnittstellen, die sie tatsächlich brauchen

    \vspace{0.4em}
    \item \textbf{Dependency Injection (Konstruktor)}\\
          Producer/Consumer erhalten \texttt{unique\_ptr<IIPCWriter/Reader>} via Konstruktor\\
          $\Rightarrow$ in Unit-Tests einfach durch Mock ersetzbar

    \vspace{0.4em}
    \item \textbf{Producer = Named-Pipe-Server}\\
          Producer legt die Pipe an und wartet auf Clients\\
          $\Rightarrow$ natürlicher Startpunkt, Consumer kann jederzeit verbinden/trennen
\end{itemize}
\end{frame}

\begin{frame}{Designentscheidungen -- Robustheit}
\begin{itemize}
    \item \textbf{Typed Exception: \texttt{PipeDisconnectedException}}\\
          Statt fragiler String-Vergleiche -- klare Fehlerklassifizierung per \texttt{catch}-Block

    \vspace{0.4em}
    \item \textbf{Reconnect-Logik im Producer}\\
          \texttt{ERROR\_BROKEN\_PIPE} $\rightarrow$ \texttt{DisconnectNamedPipe} + \texttt{ConnectNamedPipe}\\
          Producer läuft weiter; neuer Consumer kann sich jederzeit neu verbinden

    \vspace{0.4em}
    \item \textbf{Thread-Sicherheit}\\
          \texttt{std::atomic<bool> running\_} für Stop-Flag (signal-safe)\\
          \texttt{std::mutex stats\_mutex\_} schützt \texttt{Statistics}-Struktur

    \vspace{0.4em}
    \item \textbf{Signal-Handling via \texttt{IStopper}}\\
          \texttt{SIGINT}/\texttt{SIGTERM} rufen \texttt{stop()} auf dem globalen Instanz-Pointer auf\\
          $\Rightarrow$ sauberer Shutdown ohne \texttt{exit()} oder globale Flags

    \vspace{0.4em}
    \item \textbf{Timeout-Implementierung}\\
          Windows: \texttt{PeekNamedPipe}-Loop mit \texttt{GetTickCount64}\\
          Linux: \texttt{poll()} mit \texttt{struct pollfd}
\end{itemize}
\end{frame}

% ── Sprachentscheidung ──────────────────────────────────────────────────
\begin{frame}{Sprachentscheidung -- C++17 für Embedded}
\begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Bewusste Wahl: C++17}
    \begin{itemize}
        \item Interfaces, RAII, Smart Pointer\\
              $\rightarrow$ \textit{zero-overhead abstractions}
        \item \texttt{constexpr}, \texttt{[[nodiscard]]}\\
              $\rightarrow$ compile-time Prüfungen
        \item \texttt{std::atomic<bool>}\\
              $\rightarrow$ signal-sicheres Stop-Flag
        \item Dependency Injection\\
              $\rightarrow$ Komponenten testbar \& austauschbar
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Übertragbarkeit auf Embedded}
    \begin{itemize}
        \item \textbf{Embedded Linux}: vollständig einsetzbar
        \item \textbf{Bare-Metal / RTOS}:
        \begin{itemize}
            \item Heap $\rightarrow$ statische Allokation
            \item Exceptions $\rightarrow$ Fehlercodes (\texttt{-fno-exceptions})
            \item RTTI deaktiviert (\texttt{-fno-rtti})
        \end{itemize}
        \item \textbf{Architektur bleibt identisch}\\
              Factory, Interface Segregation,\\
              DI sind hardware-unabhängig
    \end{itemize}
\end{columns}
\end{frame}

\section{IPC Implementierung}

\begin{frame}[fragile]{IPC -- Abstraktion \& Plattform}
\begin{columns}[T]
    \column{0.50\textwidth}
    \textbf{Interface}
\begin{lstlisting}[style=cpp]
struct IIPCFactory {
  virtual ~IIPCFactory() = default;
  virtual unique_ptr<IIPCWriter>
    createWriter(string name) = 0;
  virtual unique_ptr<IIPCReader>
    createReader(string name) = 0;
};

// main.cpp
if (transport == "pipe")
  return make_unique<PipeIPCFactory>();
// transport == "tcp"  -- geplant
\end{lstlisting}

    \column{0.46\textwidth}
    \textbf{Plattform-Implementierung}\\[0.3em]
    \textit{Windows (Named Pipes):}
    \begin{itemize}\small
        \item \texttt{CreateNamedPipeA}
        \item \texttt{ConnectNamedPipe}
        \item \texttt{WriteFile} / \texttt{ReadFile}
        \item \texttt{PeekNamedPipe} (Timeout)
    \end{itemize}
    \vspace{0.4em}
    \textit{Linux (FIFOs):}
    \begin{itemize}\small
        \item \texttt{mkfifo}
        \item \texttt{open(O\_WRONLY)}
        \item \texttt{open(O\_RDONLY|O\_NONBLOCK)}
        \item \texttt{poll()} (Timeout)
    \end{itemize}
\end{columns}
\end{frame}

struct Event {
  time_point timestamp;
  string     component;

enum class Severity {
  INFO    = 0,
  WARNING = 1,
  ERROR   = 2

\section{Mögliche Erweiterungen}

\begin{frame}{Erweiterung -- Logging-System}
\begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Konzept: \texttt{ILogger}-Interface}
    \begin{itemize}
        \item Neue Schnittstelle analog zu \texttt{IEventSerializer}
        \item Methode \texttt{log(Severity, string)}
        \item Per Dependency Injection in Producer und Consumer injizierbar
        \item Gleiche \texttt{Severity}-Enum wie Event-Domain
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Implementierungen}
    \begin{itemize}
        \item \texttt{ConsoleLogger} -- Ausgabe auf \texttt{stdout}
        \item \texttt{FileLogger} -- persistente Logdatei
        \item \texttt{RotatingFileLogger} -- Log-Rotation nach Dateigröße
        \item \texttt{NullLogger} -- kein Output (Tests)
    \end{itemize}
    \vspace{0.3em}
    \small Austausch ohne Codeänderung an Business Logic -- selbes Muster wie Transport-Schicht
\end{columns}
\end{frame}

\begin{frame}{Weitere mögliche Erweiterungen}
\begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Neue Transportschichten}
    \begin{itemize}

        \item \textbf{Persistenz}\\
              Event-Store / Datenbank
        \item \textbf{Konfigurationsdatei}\\
              YAML/JSON statt CLI-Argumente
    \end{itemize}
\end{columns}
