\section{Zusammenfassung}

\begin{frame}{Zusammenfassung: Design- \& Architekturentscheidungen}
\begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Architektur}
    \begin{itemize}\setlength{\itemsep}{0.3em}
        \item \textbf{Abstract Factory}: Transport austauschbar (\texttt{IIPCFactory})
        \item \textbf{Interface Segregation}: möglichst schlanke Interfaces
        \item \textbf{Dependency Injection} im Konstruktor $\rightarrow$ testbar mit Fakes
        \item \textbf{Symmetrisches Binary}: Rolle via CLI-Flag
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Design}
    \begin{itemize}\setlength{\itemsep}{0.3em}
        \item \textbf{Producer = Pipe-Server}, \textbf{Consumer = Pipe-Client}
        \item \textbf{Synchrone IPC}: blockierendes \texttt{write()}, gepufferte \texttt{poll()}/\texttt{PeekNamedPipe}
        \item \textbf{Robustheit}: typisierte \texttt{PipeDisconnectedException} + Reconnect-Logik
        \item \textbf{C++17}: RAII, Smart Pointer, \texttt{std::atomic}, \texttt{constexpr}
    \end{itemize}
\end{columns}
\end{frame}

        \item \texttt{poll()} (Timeout)
