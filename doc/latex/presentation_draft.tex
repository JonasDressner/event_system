\section{Architektur}

\begin{frame}{Architekturübersicht}
\begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Hauptkomponenten}
    \begin{itemize}
        \item \texttt{main.cpp}: Entry Point, CLI-Parsing
        \item \texttt{Producer} / \texttt{Consumer}: Business Logic
        \item \texttt{IIPCFactory}: Abstract Factory (Transport-Auswahl)
        \item \texttt{IIPCWriter} / \texttt{IIPCReader}: IPC-Abstraktion
        \item \texttt{IPCWriter} / \texttt{IPCReader}: Plattform-Impl.
        \item \texttt{IEventSerializer}: Serialisierungs-Abstraktion
        \item \texttt{Event} / \texttt{Severity}: Domain-Modell
    \end{itemize}

    \column{0.45\textwidth}
    \textbf{Ablauf}
    \begin{enumerate}
        \item[\textbf{1.}] \texttt{--producer} starten \hfill \textit{\small (Nutzer)}
        \item[\textbf{2.}] Pipe anlegen, auf Consumer warten \hfill \textit{\small (automatisch)}
        \item[\textbf{3.}] \texttt{--consumer} starten \hfill \textit{\small (Nutzer)}
        \item[\textbf{4.}] Verbindung aufbauen, Eventaustausch läuft \hfill \textit{\small (automatisch)}
    \end{enumerate}
    \vspace{0.5em}
    \small Transport wählbar: \texttt{--transport pipe}\\
    \small (zukünftig: \texttt{--transport tcp})
\end{columns}
\end{frame}

\section{Designentscheidungen}

\begin{frame}{Designentscheidungen -- Interfaces \& Patterns}
\begin{itemize}\setlength{\itemsep}{0.35em}
    \item \textbf{Abstract Factory Pattern}\\
          \texttt{IIPCFactory} $\rightarrow$ \texttt{PipeIPCFactory}: neue Transportimpl.\ ohne Änderung an Business Logic

    \item \textbf{Interface Segregation}\\
          5 Interfaces: \texttt{IIPCWriter/Reader/Factory}, \texttt{IEventSerializer}, \texttt{IStopper}\\
          Klassen kennen nur die Schnittstellen, die sie tatsächlich brauchen

    \item \textbf{Dependency Injection (Konstruktor)}\\
          \texttt{unique\_ptr<IIPCWriter/Reader>}, \texttt{shared\_ptr<IEventSerializer>} via Konstruktor\\
          $\Rightarrow$ in Tests einfach durch Fakes ersetzbar

    \item \textbf{Producer = Named-Pipe-Server}\\
          Legt Pipe an, wartet auf Consumer $\Rightarrow$ Consumer kann jederzeit verbinden/trennen
\end{itemize}
\end{frame}

\begin{frame}{Designentscheidungen -- Robustheit}
\begin{itemize}\setlength{\itemsep}{0.35em}
    \item \textbf{Typed Exception: \texttt{PipeDisconnectedException}}\\
          Klare Fehlerklassifizierung per \texttt{catch}-Block statt fragiler String-Vergleiche

    \item \textbf{Reconnect-Logik im Producer}\\
          Bei Disconnect: \texttt{DisconnectNamedPipe} + \texttt{ConnectNamedPipe}\\
          Producer läuft weiter, Consumer kann sich neu verbinden

    \item \textbf{Thread-Sicherheit}\\
          \texttt{std::atomic<bool>} für Stop-Flag, \texttt{std::mutex} schützt Statistics

    \item \textbf{Signal-Handling via \texttt{IStopper}}\\
          \texttt{SIGINT}/\texttt{SIGTERM} $\Rightarrow$ \texttt{stop()} -- sauberer Shutdown ohne \texttt{exit()}

    \item \textbf{Timeout}\\
          Windows: \texttt{PeekNamedPipe}-Loop \quad Linux: \texttt{poll(struct pollfd)}
\end{itemize}
\end{frame}

% ── Sprachentscheidung ──────────────────────────────────────────────────
\begin{frame}{Sprachentscheidung -- C++17 für Embedded}
\begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Bewusste Wahl: C++17}
    \begin{itemize}
        \item Interfaces, RAII, Smart Pointer\\
              $\rightarrow$ \textit{zero-overhead abstractions}
        \item \texttt{constexpr}, \texttt{[[nodiscard]]}\\
              $\rightarrow$ compile-time Prüfungen
        \item \texttt{std::atomic<bool>}\\
              $\rightarrow$ signal-sicheres Stop-Flag
        \item Dependency Injection\\
              $\rightarrow$ Komponenten testbar \& austauschbar
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Übertragbarkeit auf Embedded}
    \begin{itemize}
        \item \textbf{Embedded Linux}: vollständig einsetzbar
        \item \textbf{Bare-Metal / RTOS}:
        \begin{itemize}
            \item Heap $\rightarrow$ statische Allokation
            \item Exceptions $\rightarrow$ Fehlercodes (\texttt{-fno-exceptions})
            \item RTTI deaktiviert (\texttt{-fno-rtti})
        \end{itemize}
        \item \textbf{Architektur bleibt identisch}\\
              Factory, Interface Segregation,\\
              DI sind hardware-unabhängig
    \end{itemize}
\end{columns}
\end{frame}

\section{IPC Implementierung}

\begin{frame}[fragile]{IPC -- Abstraktion \& Plattform}
\begin{columns}[T]
    \column{0.50\textwidth}
    \textbf{Interface}
\begin{lstlisting}[style=cpp]
struct IIPCFactory {
  virtual ~IIPCFactory() = default;
  virtual unique_ptr<IIPCWriter>
    createWriter(string name) = 0;
  virtual unique_ptr<IIPCReader>
    createReader(string name) = 0;
};

// main.cpp
if (transport == "pipe")
  return make_unique<PipeIPCFactory>();
// transport == "tcp"  -- geplant
\end{lstlisting}

    \column{0.46\textwidth}
    \textbf{Plattform-Implementierung}\\[0.3em]
    \textit{Windows (Named Pipes):}
    \begin{itemize}\small
        \item \texttt{CreateNamedPipeA}
        \item \texttt{ConnectNamedPipe}
        \item \texttt{WriteFile} / \texttt{ReadFile}
        \item \texttt{PeekNamedPipe} (Timeout)
    \end{itemize}
    \vspace{0.4em}
    \textit{Linux (FIFOs):}
    \begin{itemize}\small
        \item \texttt{mkfifo}
        \item \texttt{open(O\_WRONLY)}
        \item \texttt{open(O\_RDONLY|O\_NONBLOCK)}
        \item \texttt{poll()} (Timeout)
    \end{itemize}
\end{columns}
\end{frame}

\section{Mögliche Erweiterungen}

\begin{frame}[fragile]{Erweiterung -- Logging-System}
\begin{columns}[T]
    \column{0.50\textwidth}
    \textbf{Interface (analog zu \texttt{IEventSerializer})}
\begin{lstlisting}[style=cpp]
struct ILogger {
  virtual ~ILogger() = default;
  virtual void log(Severity sev,
                   const string& msg) = 0;
};
\end{lstlisting}
    \vspace{0.2em}
    \textbf{Einbindung per Dependency Injection}
\begin{lstlisting}[style=cpp]
Producer::Producer(
  unique_ptr<IIPCWriter> writer,
  shared_ptr<IEventSerializer> ser,
  shared_ptr<ILogger> logger)  // neu
  : logger_(std::move(logger)) { ... }
\end{lstlisting}

    \column{0.47\textwidth}
    \textbf{Verwendung im Code}
\begin{lstlisting}[style=cpp]
// Producer::start()
logger_->log(Severity::INFO,
  "Event sent: " + evt.component);

// Consumer::processEvent()
logger_->log(evt.severity,
  evt.component + ": " + evt.message);
\end{lstlisting}
    \vspace{0.3em}
    \textbf{Implementierungen}
    \begin{itemize}\small
        \item \texttt{ConsoleLogger}: \texttt{stdout}
        \item \texttt{FileLogger}: persistente Datei
        \item \texttt{NullLogger}: kein Output (Tests)
    \end{itemize}
    \vspace{0.2em}
    \small\textit{Kein \texttt{if/switch} nötig, gleiches Muster wie Transport-Schicht}
\end{columns}
\end{frame}
